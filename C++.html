 // C++ code

int num = 0;

void setup()
{
  pinMode(A0, INPUT);
  pinMode(12, OUTPUT);

  num = analogRead(A0);
}

void loop()
{
  num = 0;
  num += analogRead(A0);
  if (num >= 500) {
    digitalWrite(12, HIGH);
  } else {
    digitalWrite(12, LOW);
  }
  delay(500); // Wait for 500 millisecond(s)
}
